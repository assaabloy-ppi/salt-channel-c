  cmocka

  GIT_TAG "master"
  UPDATE_COMMAND ""
  INSTALL_COMMAND ""
  DOWNLOAD_DIR ""
  CMAKE_ARGS "-DCMAKE_INSTALL_PREFIX=${CMAKE_BINARY_DIR}/external/cmocka"
  INSTALL_DIR ${CMAKE_BINARY_DIR}/external/cmocka
  INSTALL_COMMAND
  "cmake"
  --build .
  --target install
)

add_library(cmocka_lib SHARED IMPORTED)

add_dependencies( cmocka_lib cmocka )


macro(do_test arg)
    add_definitions(-std=c99 -Wno-unused-parameter)
    add_executable(${arg} ${arg}.c)
    add_sanitizers(${arg})
    add_dependencies(${arg} ${arg} ${ARGN} crypto_api)
    add_test(${arg} ${arg})
    target_link_libraries(${arg} ${ARGN} crypto_api cmocka_lib ${EXTRA_LIBS})
endmacro(do_test)
